#include <iostream>
#include <fstream>

using namespace std;

int L = 0, R = 0;

int main(int argc, char **argv)
{
	string str = argv[1];
	//char *str = argv[1];
	fstream fs(str);
	while (!fs.eof())
	{
		getline(fs, str);
		//fs.getline(str, 100); 	效果类似
		if ('<' == str[0])
			L++;
		else if ('>' == str[0])
			R++;
	}

	cout << "R: " << R << " L: " << L << endl;
	return 0;
}

getline函数，第一个参数为文件对象流，第二个参数为字符串类型(string)，将文件对象流中的一行字符读到string里，同时，让文件对象流的位置移到下一行。

fstream的构造函数中第二个参数可以填写打开模式，有fstream::in, fstream::out等6种模式。
fstream对象也实现了getline函数，第一个参数为字符数组的指针，第二个参数为字符数组大小，返回类型为原对象的引用(即可以多个getline连续使用)。
fstream对象的is_open()判断是否打开文件，返回1为正常打开。
fstream对象的eof()判断是否到文件结尾，返回1为文件结束。
对于windows下的文件路径，无论在输入路径时是否使用转义字符(即\和\\)，系统都会以正确的格式读取路径。
